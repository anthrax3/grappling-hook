Examples

instance.unhook('pre:save', onPreSave);
//removes all middleware for `pre:save`
instance.unhook('pre:save');
//removes all middleware for `pre:save` and `post:save`
instance.unhook('save');
//removes ALL middleware
instance.unhook();
